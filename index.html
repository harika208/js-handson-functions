7. Write a function that accepts parameter n and returns factorial of n 
    function fact(n){
        let ans = 1;
        if(n == 0 || n == 1){
            return ans;
        }
        else{
            for(var i = n; i >=1; i--){
                ans = ans * i;
            }
            return ans;
        }
    }
    let n = 4;
    ans = fact();
    console.log("factorial of " +n+ " is " +ans);

    OUTPUT:- factorial of 4 is 1
